### **Step 4: Core Transaction Logic (Backend)**
```typescript
await prisma.$transaction(async (tx) => {
  // 1. Check inventory
  const inventory = await tx.bloodInventory.findUnique({...});
  
  // 2. Determine fulfillment
  if (inventory && inventory.quantity >= quantityNeeded) {
    // ✅ Sufficient: FULFILL + REDUCE inventory
    await tx.bloodInventory.update({
      data: { quantity: { decrement: quantityNeeded } }
    });
    status = "FULFILLED";
  } else {
    // ❌ Insufficient: PENDING + NO inventory change
    status = "PENDING";
  }
  
  // 3. Create request
  const request = await tx.bloodRequest.create({
    data: { status, ...otherFields }
  });
  
  // ⚛️ ATOMIC: All operations succeed or all rollback
});
```
